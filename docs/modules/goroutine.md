                    local error = result.values[4]
                    log.error("‚ùå " .. name .. ": " .. status .. " - " .. error)
                end
            else
                unhealthy_count = unhealthy_count + 1
                log.error("‚ùå Error: " .. (result.error or "Unknown error"))
            end
        end
        
        log.info("‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê")
        log.info("üìä Summary: " .. healthy_count .. " healthy, " .. unhealthy_count .. " unhealthy")
        
        return unhealthy_count == 0, "Health check completed", {
            total = #services,
            healthy = healthy_count,
            unhealthy = unhealthy_count
        }
    end)
    :timeout("30s")
    :build()

workflow.define("health_check_workflow")
    :description("Parallel health check for multiple services")
    :version("1.0.0")
    :tasks({ parallel_health_check })
