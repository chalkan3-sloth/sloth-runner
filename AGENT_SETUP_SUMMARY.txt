🚀 COMO INICIAR UM AGENT:
═══════════════════════════════════════════════════════════════════════════

  1️⃣  INICIE O MASTER (uma vez, nesta máquina)
      
      cd /Users/chalkan3/.projects/task-runner
      ./start_master.sh

  2️⃣  INICIE AGENT LOCAL (para teste)
      
      ./start_local_agent.sh meu-agent

  3️⃣  INICIE AGENTS REMOTOS
      
      # Para ladyguica (192.168.1.16)
      ./manage_remote_agent.sh start chalkan3@192.168.1.16 \
        ladyguica 192.168.1.16

      # Para keiteguica (192.168.1.17)
      ./manage_remote_agent.sh start chalkan3@192.168.1.17 \
        keiteguica 192.168.1.17

  4️⃣  VERIFIQUE OS AGENTS
      
      ./sloth-runner agent list --master 192.168.1.29:50053

  5️⃣  USE EM SCRIPTS LUA
      
      local t = task("minha_tarefa")
          :delegate_to("ladyguica")  -- Executa no agent remoto
          :command(function(this, params)
              local exec = require("exec")
              return exec.run("hostname")
          end)
          :build()
      
      t:run()

📋 COMANDOS ESSENCIAIS:
═══════════════════════════════════════════════════════════════════════════

  # Listar agents
  ./sloth-runner agent list --master 192.168.1.29:50053

  # Executar comando remoto
  ./sloth-runner agent run \
    --master 192.168.1.29:50053 \
    --agent ladyguica \
    --command "hostname"

  # Status de agent remoto
  ./manage_remote_agent.sh status chalkan3@192.168.1.16 ladyguica

  # Parar agent remoto
  ./manage_remote_agent.sh stop chalkan3@192.168.1.16 ladyguica

  # Ver processos
  ps aux | grep sloth-runner

  # Parar master
  pkill -f "sloth-runner master"
